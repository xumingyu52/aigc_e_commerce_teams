### 1.1 Hooks设计缺陷

#### 问题1：useStream Hook - 缺少AbortController

**缺陷**：
```typescript
// ❌ 问题：无法取消正在进行的请求
const startStream = useCallback(async (endpoint: string, body: object) => {
  // 没有AbortController，组件卸载时请求仍在继续
}, []);
```

**修复**：
```typescript
// ✅ 修复：添加AbortController支持
export function useStream() {
  const [content, setContent] = useState('');
  const [isStreaming, setIsStreaming] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const abortControllerRef = useRef<AbortController | null>(null);
  
  const startStream = useCallback(async (endpoint: string, body: object) => {
    // 取消之前的请求
    if (abortControllerRef.current) {
      abortControllerRef.current.abort();
    }
    
    const controller = new AbortController();
    abortControllerRef.current = controller;
    
    setIsStreaming(true);
    setContent('');
    setError(null);
    
    try {
      const token = localStorage.getItem('token');
      const response = await fetch(endpoint, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          ...(token && { Authorization: `Bearer ${token}` })
        },
        body: JSON.stringify(body),
        signal: controller.signal // 添加signal
      });
      
      if (!response.ok) {
        throw new Error(`请求失败: ${response.status}`);
      }
      
      const reader = response.body?.getReader();
      if (!reader) {
        throw new Error('无法读取响应流');
      }
      
      const decoder = new TextDecoder();
      
      while (true) {
        // 检查是否被取消
        if (controller.signal.aborted) {
          break;
        }
        
        const { done, value } = await reader.read();
        if (done) break;
        
        const chunk = decoder.decode(value, { stream: true });
        const lines = chunk.split('\n');
        
        for (const line of lines) {
          if (line.startsWith('data:')) {
            try {
              const data = JSON.parse(line.slice(5));
              if (data.chunk) {
                setContent(prev => prev + data.chunk);
              }
              if (data.error) {
                throw new Error(data.error);
              }
            } catch (e) {
              // 忽略JSON解析错误，可能是部分数据
            }
          }
        }
      }
    } catch (err) {
      if (err instanceof Error && err.name === 'AbortError') {
        // 请求被取消，不显示错误
        return;
      }
      const message = err instanceof Error ? err.message : '流式请求失败';
      setError(message);
      toast.error(message);
    } finally {
      setIsStreaming(false);
      abortControllerRef.current = null;
    }
  }, []);
  
  // 组件卸载时取消请求
  useEffect(() => {
    return () => {
      if (abortControllerRef.current) {
        abortControllerRef.current.abort();
      }
    };
  }, []);
  
  const reset = useCallback(() => {
    setContent('');
    setError(null);
  }, []);
  
  return { content, isStreaming, error, startStream, reset };
}
```

**缺陷**：
```typescript

const { data, error, isLoading, mutate } = useSWR(
  ['dashboard', params], // params对象引用可能变化

  { ... }

```

**修复**：
```typescript
// ✅ 修复：使用序列化的key
import useSWR from 'swr';

export function useDashboard(params: DashboardParams) {
  // 使用JSON.stringify生成稳定的key
  const key = useMemo(
    () => ['dashboard', JSON.stringify(params)],
    [params.timeRange, params.type, params.platforms?.join(',')]
  );
  
  const { data, error, isLoading, mutate } = useSWR(
    key,
    () => getDashboardData(params),
    {
      refreshInterval: 60000,
      revalidateOnFocus: false,
      dedupingInterval: 10000,
      errorRetryCount: 3, // 添加重试
      errorRetryInterval: 5000,
      onErrorRetry: (error, key, config, revalidate, { retryCount }) => {
        // 404不重试
        if (error.status === 404) return;
        // 最多重试3次
        if (retryCount >= 3) return;
        // 5秒后重试
        setTimeout(() => revalidate({ retryCount }), 5000);
      }
    }
  );
  
  return {
    data,
    error,
    isLoading,
    refresh: mutate
  };
}
```

#### 问题3：useTaskPolling Hook - 缺少轮询清理

**缺陷**：
```typescript
// ❌ 问题：轮询可能在组件卸载后继续
useEffect(() => {
  if (!taskId || status === 'completed' || status === 'failed') return;
  
  const interval = setInterval(async () => {
    // 轮询逻辑
  }, 2000);
  
  return () => clearInterval(interval); // 虽然有清理，但没有指数退避
}, [taskId, status]);
```

**修复**：
```typescript
// ✅ 修复：添加指数退避和更好的清理
export function useTaskPolling() {
  const [taskId, setTaskId] = useState<string | null>(null);
  const [status, setStatus] = useState<TaskStatus>('idle');
  const [progress, setProgress] = useState(0);
  const [result, setResult] = useState<TaskResult | null>(null);
  const [logs, setLogs] = useState<string[]>([]);
  const intervalRef = useRef<NodeJS.Timeout | null>(null);
  const retryCountRef = useRef(0);
  
  const stopPolling = useCallback(() => {
    if (intervalRef.current) {
      clearInterval(intervalRef.current);
      intervalRef.current = null;
    }
  }, []);
  
  const startPolling = useCallback((id: string) => {
    stopPolling();
    retryCountRef.current = 0;
    
    const poll = async () => {
      try {
        const data = await aiToolsApi.getTaskStatus(id);
        setStatus(data.task.status);
        setProgress(data.task.progress || 0);
        setResult(data.task.result || null);
        setLogs(data.task.logs || []);
        
        // 重置重试计数
        retryCountRef.current = 0;
        
        if (data.task.status === 'completed' || data.task.status === 'failed') {
          stopPolling();
        }
      } catch (err) {
        retryCountRef.current++;
        
        // 指数退避：2s, 4s, 8s, 最大30s
        const delay = Math.min(2000 * Math.pow(2, retryCountRef.current), 30000);
        
        if (retryCountRef.current >= 5) {
          stopPolling();
          setStatus('failed');
          return;
        }
        
        // 重新设置间隔
        stopPolling();
        intervalRef.current = setInterval(poll, delay);
      }
    };
    
    intervalRef.current = setInterval(poll, 2000);
  }, [stopPolling]);
  
  // 组件卸载时清理
  useEffect(() => {
    return () => {
      stopPolling();
    };
  }, [stopPolling]);
  
  const submitTask = useCallback(async (type: 'image' | 'video', params: object) => {
    setStatus('pending');
    setProgress(0);
    setResult(null);
    setLogs([]);
    
    const data = type === 'image'
      ? await aiToolsApi.generateImage(params.imageUrl)
      : await aiToolsApi.generateVideo(params.imageUrl, params.text);
    
    setTaskId(data.task_id);
    startPolling(data.task_id);
    
    return data.task_id;
  }, [startPolling]);
  
  const reset = useCallback(() => {
    stopPolling();
    setTaskId(null);
    setStatus('idle');
    setProgress(0);
    setResult(null);
    setLogs([]);
    retryCountRef.current = 0;
  }, [stopPolling]);
  
  return { taskId, status, progress, result, logs, submitTask, reset };
}
```

#### 问题4：Promise.all容错性不足

**缺陷**：
```typescript
// ❌ 问题：一个请求失败，全部失败
const [revenueData, profitData, platformData, productSales] = await Promise.all([
  fetchRevenueData(params),
  fetchProfitData(params),
  fetchPlatformData(params),
  fetchProductSales(params)
]);
```

**修复**：
```typescript
// ✅ 修复：使用Promise.allSettled提供容错
export async function getDashboardData(params: DashboardParams) {
  const results = await Promise.allSettled([
    fetchRevenueData(params),
    fetchProfitData(params),
    fetchPlatformData(params),
    fetchProductSales(params)
  ]);
  
  // 处理每个结果，失败的返回null或默认值
  const [revenueResult, profitResult, platformResult, productSalesResult] = results;
  
  return {
    revenue: revenueResult.status === 'fulfilled' ? revenueResult.value : null,
    profit: profitResult.status === 'fulfilled' ? profitResult.value : null,
    platforms: platformResult.status === 'fulfilled' ? platformResult.value : [],
    productSales: productSalesResult.status === 'fulfilled' ? productSalesResult.value : [],
    // 记录失败的请求
    errors: results
      .filter((r): r is PromiseRejectedResult => r.status === 'rejected')
      .map(r => r.reason)
  };
}
```

#### 问题5：HTTP客户端缺少请求取消

**缺陷**：
```typescript
// ❌ 问题：没有请求取消机制
export async function request<T>(endpoint: string, options?: RequestInit): Promise<T> {
  const response = await fetch(endpoint, options);
  return response.json();
}
```

**修复**：
```typescript
// ✅ 修复：添加AbortController支持
const pendingRequests = new Map<string, AbortController>();

export async function request<T>(
  endpoint: string,
  options?: RequestInit & { requestId?: string }
): Promise<T> {
  const { requestId, ...fetchOptions } = options || {};
  
  // 如果有requestId，取消之前的相同请求
  if (requestId && pendingRequests.has(requestId)) {
    pendingRequests.get(requestId)?.abort();
  }
  
  const controller = new AbortController();
  
  if (requestId) {
    pendingRequests.set(requestId, controller);
  }
  
  try {
    const token = localStorage.getItem('token');
    const response = await fetch(endpoint, {
      ...fetchOptions,
      signal: controller.signal,
      headers: {
        'Content-Type': 'application/json',
        ...(token && { Authorization: `Bearer ${token}` }),
        ...fetchOptions.headers
      }
    });
    
    if (!response.ok) {
      throw new Error(`API Error: ${response.status} ${response.statusText}`);
    }
    
    return response.json();
  } finally {
    if (requestId) {
      pendingRequests.delete(requestId);
    }
  }
}

// 取消特定请求
export function cancelRequest(requestId: string) {
  pendingRequests.get(requestId)?.abort();
  pendingRequests.delete(requestId);
}
```

```typescript
export const BaseChart = memo(function BaseChart({
  option,
  className = '',
  height = 400
}: BaseChartProps) {

        notMerge={false}
        lazyUpdate={true}
        style={{ height: '100%', width: '100%' }}
```

## 二、优化后的完整Hooks实现

### 2.1 useAuth Hook（优化版）

```typescript
// lib/hooks/use-auth.ts
import { useState, useEffect, useCallback, useMemo } from 'react';
import { useRouter } from 'next/navigation';
import { authApi } from '@/lib/api/auth';
import { authStore } from '@/lib/stores/auth-store';
import type { User } from '@/lib/types/user';

export function useAuth() {
  const router = useRouter();
  const [user, setUser] = useState<User | null>(null);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  
  // 初始化时检查token
  useEffect(() => {
    const checkAuth = async () => {
      const token = localStorage.getItem('token');
      if (!token) {
        setIsLoading(false);
        return;
      }
      
      try {
        const data = await authApi.me();
        setUser(data.user);
        authStore.getState().setUser(data.user);
      } catch (err) {
        // Token无效，清除
        localStorage.removeItem('token');
        setError('认证失败，请重新登录');
      } finally {
        setIsLoading(false);
      }
    };
    
    checkAuth();
  }, []);
  
  const login = useCallback(async (email: string, password: string) => {
    setError(null);
    try {
      const data = await authApi.login(email, password);
      localStorage.setItem('token', data.token);
      
      const userData = await authApi.me();
      setUser(userData.user);
      authStore.getState().setUser(userData.user);
      
      router.push('/home');
      return { success: true };
    } catch (err) {
      const message = err instanceof Error ? err.message : '登录失败';
      setError(message);
      return { success: false, error: message };
    }
  }, [router]);
  
  const logout = useCallback(async () => {
    try {
      await authApi.logout();
    } catch (err) {
      // 忽略登出错误
    } finally {
      localStorage.removeItem('token');
      setUser(null);
      authStore.getState().clearUser();
      router.push('/');
    }
  }, [router]);
  
  // 使用useMemo稳定返回值
  const auth = useMemo(() => ({
    user,
    isLoading,
    error,
    isAuthenticated: !!user,
    login,
    logout
  }), [user, isLoading, error, login, logout]);
  
  return auth;
}
```

### 2.2 useProducts Hook（优化版）

```typescript
// lib/hooks/use-products.ts
import { useMemo } from 'react';
import useSWR from 'swr';
import useSWRMutation from 'swr/mutation';
import { productsApi } from '@/lib/api/products';
import type { Product, CreateProductDto, UpdateProductDto } from '@/lib/types/product';

interface UseProductsParams {
  category?: string;
  page?: number;
  pageSize?: number;
}

export function useProducts(params: UseProductsParams = {}) {
  // 稳定的key
  const key = useMemo(
    () => ['products', params.category || 'all', params.page || 1],
    [params.category, params.page]
  );
  
  const { data, error, isLoading, mutate } = useSWR(
    key,
    () => productsApi.list(params),
    {
      revalidateOnFocus: false,
      dedupingInterval: 5000
    }
  );
  
  // 创建商品
  const { trigger: createProduct, isMutating: isCreating } = useSWRMutation(
    'products',
    async (_key, { arg }: { arg: CreateProductDto }) => {
      const result = await productsApi.create(arg);
      await mutate(); // 刷新列表
      return result;
    }
  );
  
  // 更新商品
  const { trigger: updateProduct, isMutating: isUpdating } = useSWRMutation(
    'products',
    async (_key, { arg }: { arg: { id: string; data: UpdateProductDto } }) => {
      const result = await productsApi.update(arg.id, arg.data);
      await mutate();
      return result;
    }
  );
  
  // 删除商品
  const { trigger: deleteProduct, isMutating: isDeleting } = useSWRMutation(
    'products',
    async (_key, { arg }: { arg: string }) => {
      const result = await productsApi.delete(arg);
      await mutate();
      return result;
    }
  );
  
  return {
    products: data?.data || [],
    total: data?.total || 0,
    error,
    isLoading,
    createProduct,
    isCreating,
    updateProduct,
    isUpdating,
    deleteProduct,
    isDeleting,
    refresh: mutate
  };
}
```
